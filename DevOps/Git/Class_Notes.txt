	GIT Cli Commands :
	
	
	- git clone			# To Copy/Clone the entire Remote Repository to the Local Machine

	- git add 			# To Add the Changes from working directory to the Staging Area 
	
	- git commit 		# To Commit the Changes from the Staging Area to Local Repository
	
	- git push 			# To push the changes to from Local Repository to Remote Repository
	
	- git pull/fetch	# Both git pull/fetch are used to handle the incremental changes from Remote Repository
	
	
		git fetch:		
				# Used to just check for the  incremental changes from remote repository, if any incremental changes exist, it will update the details in the Local Repository. Git Fetch will never update the working directory.
				
				Perform explicit merge to working directory using branching techniques.
				
				
		git pull:
				# Used to check for the  incremental changes from remote repository, if any incremental changes exist, it will update the Local Repository as well the working directory.
				
				
				git pull ==> git fetch + git merge
				
				# As a best practise, it is always recommended to use git pull to keep the local repository in-sync with remote repository
				
	
	- Fork : 
				# To copy one remote repository to another remote repository
				
	
	
	- git init:
				# To initialize a local repository
				# It create .git/ Directory 
				# It creates the master/main branch
